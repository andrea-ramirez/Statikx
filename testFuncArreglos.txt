func int -> fact(int -> j){
    var int -> i;

    i = j + (p-j*2+j);

    if(j == 1)
    True{
        returns j;
    }False{
        returns j*fact(j-1);;
    }
}

func void -> inicia(int -> y){
    var int -> x;

    x = 1;

    while(x < 11){
        Arreglo[x] = y * x;
        x = x + 1;
    }
}
